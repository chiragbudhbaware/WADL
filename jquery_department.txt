// Get URL Parameter

const params =
new URLSearchParams(window.location.search);

const dept = params.get("dept");


// Department Data

const data = {

    IT:
    "Information Technology department focuses on software development, networking and databases.",

    COMP:
    "Computer Engineering department focuses on programming, AI and data structures.",

    ETC:
    "Electronics and Telecommunication department focuses on communication systems and electronics.",

    MECH:
    "Mechanical Engineering department focuses on machines, manufacturing and thermodynamics.",

    ELECT:
    "Electrical Engineering department focuses on power systems and electrical circuits."

};


// Display Data

if(document.getElementById("deptName"))
{
    document.getElementById("deptName")
    .innerText = dept;

    document.getElementById("deptInfo")
    .innerText = data[dept];
}
